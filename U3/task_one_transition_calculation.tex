\newcommand{\ds}{$\delta^*$}
\newcommand{\ver[3]}{$\{#2\}\cup\{#3\}$}
\begin{align*}
	\delta^*(\{d,b\},0)&=\ds(\emptyset\cup\{d\},\lambda)=\{d\}\\
	\delta^*(\{d,b\},1)&=\ds(\{c\}\cup\emptyset,\lambda)=\{c\}\\
	\delta^*(\{b,c,d\},0)&=\ds(\ver{d}{b}\cup\emptyset,\lambda)=\{d,b\}\\
	\delta^*(\{b,c,d\},1)&=\ds(\emptyset\cup\emptyset\cup\{c\})\\
\end{align*}
